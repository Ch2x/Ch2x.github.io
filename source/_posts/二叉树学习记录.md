## 二叉树的前序遍历

```javascript
var preorderTraversal = function (root) {

    let p = root,
        arr = [],
        res = [];

    while (p || arr.length > 0) {
        res.push(p.val);
        if (p.right) {
            arr.push(p.right);
        }

        if (p.left) {
            arr.push(p.left);
        }
        p = arr.pop();
    }

    return res;
};
```

## 二叉树的中序遍历

```javascript
var inorderTraversal = function (root) {
    let p = root,
        arr = [],
        res = [];

    while (p || arr.length > 0) {
        while (p) {
            arr.push(p);
            p = p.left;
        }
        p = arr.pop();
        res.push(p.val);
        p = p.right;
    }

    return res;
};

```

## 二叉树的后序遍历

```javascript
vvar postorderTraversal = function (root) {
    if (!root) {
        return [];
    }

    let arr = [root],
        res = [];

    while (arr.length > 0) {
        let temp = arr.pop();
        res.unshift(temp.val);

        if (temp.left) {
            arr.push(temp.left);
        }

        if (temp.right) {
            arr.push(temp.right);
        }
    }

    return res;
};
```
