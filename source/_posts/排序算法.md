## 冒泡排序

```js
function bubbleSort(nums) {
    for (let i = 1; i < nums.length; i++) {
        for (let j = 0; j < nums.length - i; j++) {
            if (nums[j] > nums[j + 1]) {
                const temp = nums[j];
                nums[j] = nums[j + 1];
                nums[j + 1] = temp;
            }
        }
    }
    return nums;
}
```

## 选择排序

```js
function selectSort(nums) {
    for (let i = 0; i < nums.length; i++) {
        let minIdx = i;
        for (let j = i; j < nums.length; j++) {
            if (nums[j] < nums[minIdx]) {
                minIdx = j;
            }
        }
        const temp = nums[i];
        nums[i] = nums[minIdx];
        nums[minIdx] = temp;
    }
    return nums;
}
```

## 插入排序

```js
function insertSort(nums) {
    for (let i = 1; i < nums.length; i++) {
        let temp = nums[i];
        for (let j = i; j >= 0; j--) {
            if (j > 0 && nums[j - 1] > temp) {
                nums[j] = nums[j - 1];
            } else {
                nums[j] = temp;
                break;
            }
        }
    }
    return nums;
}
```

## 复杂度

排序类型 | 时间复杂度 | 空间复杂度
------- | --------- | ----
冒泡排序 | O(n2) | O(1)
选择排序 | O(n2) | O(1)
插入排序 | O(n2) | O(1)
